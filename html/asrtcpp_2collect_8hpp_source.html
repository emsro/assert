87 {

89 srv,
90 root_id,
91 timeout,
92 +[]( void* p, enum asrt_status s ) {
93 auto& self = *static_cast< op* >( p );
94 if ( s == ASRT_SUCCESS )
95 self.recv.set_value();
96 else
97 self.recv.set_error( s );
98 },
99 this );
100 if ( s != ASRT_SUCCESS )
101 recv.set_error( s );
102 }
103 };

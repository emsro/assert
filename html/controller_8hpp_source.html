99struct cntr_start_sender
100{
101 using sender_concept = ecor::sender_t;
102 using completion_signatures =
103 ecor::completion_signatures< ecor::set_value_t(), ecor::set_error_t( status ) >;
104
105 asrt_controller* c;
106 uint32_t timeout;
107
108 template < ecor::receiver R >
109 struct op
110 {
111 asrt_controller* c;
112 uint32_t timeout;
113 R recv;
114
115 void start()
116 {
117 auto s = asrt_cntr_start(
118 c,
119 +[]( void* p, asrt_status s ) -> asrt_status {
120 auto& self = *static_cast< op* >( p );
121 if ( s != ASRT_SUCCESS )
122 self.recv.set_error( s );
123 else
124 self.recv.set_value();
125 return ASRT_SUCCESS;
126 },
127 this,
128 timeout );
129 if ( s != ASRT_SUCCESS )
130 recv.set_error( s );
131 }
132 };
133
134 template < ecor::receiver R >
135 auto connect( R&& r ) && noexcept
136 {
137 return op< R >{ c, timeout, (R&&) r };
138 }
139};

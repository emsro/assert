106 template < ecor::receiver R >
107 struct op
108 {
109 collect_append_sender s;
110 R recv;
111
112 static void on_done( void* ptr, enum asrt_status st )
113 {
114 auto* o = static_cast< op* >( ptr );
115 if ( st == ASRT_SUCCESS )
116 o->recv.set_value();
117 else
118 o->recv.set_error( static_cast< status >( st ) );
119 }
120
121 void start()
122 {
123 using traits = collect_append_traits< T >;
124 using member_type = decltype( asrt_flat_scalar{}.*traits::member );
125 asrt_flat_value v = { .type = traits::flat_type };
126 v.data.s.*traits::member = static_cast< member_type >( s.val );
127 auto st = asrt_collect_client_append(
128 s.client, s.parent, s.key, &v, NULL, on_done, this );
129 if ( st != ASRT_SUCCESS )
130 recv.set_error( st );
131 }
132 };

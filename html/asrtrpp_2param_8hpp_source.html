140template < has_param_query_traits T, typed_param_query_callable< T > CB >
141ASRT_NODISCARD asrt_status
142query( ref< asrt_param_client > cl, asrt_param_query* q, flat_id node_id, char const* key, CB& cb )
143{
144 using traits = param_query_traits< T >;
145 q->expected_type = traits::flat_type;
146 q->cb.*traits::cb_member =
147 []( asrt_param_client* c, asrt_param_query* qq, traits::raw_type raw ) {
148 ( *reinterpret_cast< CB* >( qq->cb_ptr ) )(
149 c, qq, static_cast< traits::value_type >( raw ) );
150 };
151 q->cb_ptr = &cb;
152 return asrt_param_client_query( q, cl, node_id, key );
153}

278struct param_query_sender
279{
280 using sender_concept = ecor::sender_t;
281
282 using completion_signatures = ecor::completion_signatures<
283 ecor::set_value_t( param_result< T > ),
284 ecor::set_error_t( status ) >;
285
286 asrt_param_client* client;
287 char const* key;
288 flat_id node_id;

290 template < ecor::receiver R >
291 struct op
292 {
293 asrt_param_query q = {};
294 asrt_param_client* c;
295 char const* key;
296 flat_id node_id;
297 R recv;
298
299 op( R&& r, asrt_param_client* client, char const* k, flat_id id )
300 : c( client )
301 , key( k )
302 , node_id( id )
303 , recv( (R&&) r )
304 {
305 }
306
307 void start()
308 {
309 using traits = param_query_traits< T >;
310 auto cb =
311 +[]( asrt_param_client*, asrt_param_query* q, traits::raw_type raw ) {
312 auto& self = *reinterpret_cast< op* >( q->cb_ptr );
313 if ( q->error_code != ASRT_PARAM_ERR_NONE )
314 self.recv.set_error( ASRT_RECV_ERR );
315 else
316 self.recv.set_value( param_result< T >{
317 static_cast< traits::value_type >( raw ),
318 q->key,
319 q->next_sibling } );
320 };
321 auto s = query< T >( c, &q, node_id, key, cb, this );
322 if ( s != ASRT_SUCCESS )
323 recv.set_error( ASRT_RECV_ERR );
324 }
325 };

329 auto connect( R&& r ) && noexcept
330 {
331 return op< R >{ (R&&) r, client, key, node_id };
332 }
